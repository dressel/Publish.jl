%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% julia_listings.tex
% Creates a listings definition for Julia
% Should handle all the listings stuff
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\usepackage{listings}

\lstdefinelanguage{Julia}
{
	morekeywords={
		elseif, else, end, for, function, if, using, while
	},
	sensitive=true,      % case sensitive
	morecomment=[l]{\#}, % comments
	morestring=[b]"
}


\lstdefinestyle{JuliaStyle}
{
	language=Julia,
	basicstyle=\ttfamily,
	keywordstyle=\bfseries\color{blue},
	commentstyle=\color{ForestGreen},
	stringstyle=\color{magenta},
	tabsize=4,
	breaklines=true,
	xleftmargin=0.5in,
	showstringspaces=false % If true, underlines spaces in strings
}

\lstdefinestyle{Output}
{
	basicstyle=\itshape,
	xleftmargin=1.0in,
	literate={``}{\dbllq}1 {''}{\dblrq}1
}
